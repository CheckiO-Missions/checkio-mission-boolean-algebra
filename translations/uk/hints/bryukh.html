def boolean(x: bool, y: bool, op: str) -> bool:
    if op == "conjunction":
        return x and y
    if op == "disjunction":
        return x or y
    if op == "implication":
        return not (x) or y
    if op == "exclusive":
        return x ^ y
    if op == "equivalence":
